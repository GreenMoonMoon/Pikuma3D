cmake_minimum_required(VERSION 3.20)

project(Pikuma3D)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED true)


find_package(SDL2 REQUIRED PATHS external/SDL2/lib/cmake)
include_directories(${SDL2_INCLUDE_DIRS})


add_executable(pikuma3d src/main.c)
target_link_libraries(pikuma3d ${SDL2_STATIC_LIBRARIES})
